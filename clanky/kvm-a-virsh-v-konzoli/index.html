KVM a virsh v konzoli

19. November 2020 • Dodoslav
kvm virsh virtualization
navod na pouzivanie KVM cez konzolu

#!/bin/bash -x #fill values to your need VMNAME=aaaa MAC='52:54:00:09:11:03' VNC_PORT=5952 VNC_PASSWORD='!PASSWORD!' DISK='/path/to/filename.qcow2' sudo virt-install \ --connect qemu:///system \ --import \ --name=$VMNAME \ --ram=2048 \ --vcpus=2 \ --disk $DISK,device=disk,bus=virtio,format=qcow2 \ --os-variant rhel6 \ --debug \ --network=bridge:br0,mac=$MAC \ --graphics=vnc,port=$VNC_PORT,password=$VNC_PASSWORD \ --autostart \ --noautoconsole \ --os-type=linux virsh list --all virsh undefine $VM virsh destroy $VM
