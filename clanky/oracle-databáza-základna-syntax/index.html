Oracle databáza - základna syntax

Jan 1, 0001

SELECT table_name, owner FROM user_tables;

SELECT table_name, owner FROM all_tables ORDER BY owner, table_name;

select username from dba_users;

CREATE USER books_admin IDENTIFIED BY MyPassword; GRANT CONNECT TO books_admin; GRANT CONNECT, RESOURCE, DBA TO books_admin; GRANT CREATE SESSION GRANT ANY PRIVILEGE TO books_admin; GRANT UNLIMITED TABLESPACE TO books_admin;

SQL> select USERNAME,ACCOUNT_STATUS from dba_users where ORACLE_MAINTAINED <> ‘Y’;

SQL> select USERNAME,ACCOUNT_STATUS from dba_users where USERNAME like ‘%XM%’;

SQL> r 1* SELECT username, status, osuser, process, machine, terminal, logon_time FROM v$session where MACHINE <> ‘hostname’
